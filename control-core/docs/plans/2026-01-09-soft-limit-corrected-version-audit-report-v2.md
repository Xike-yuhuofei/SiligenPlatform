// ⚠️ 当前实现可能有阻塞风险
for (short axis : config_.monitored_axes) {
    auto result = motion_control_port_->StopAxis(axis);
    // 如果某个轴停止失败，会继续尝试其他轴，可能累积延迟
}

// ✅ 建议添加超时和并发停止
Result<void> StopMotion(bool use_emergency_stop) noexcept {
    if (use_emergency_stop) {
        return motion_control_port_->EmergencyStopAll();  // 一次调用
    } else {
        // 并发停止所有轴，或添加总超时
        auto start = std::chrono::steady_clock::now();
        for (short axis : config_.monitored_axes) {
            auto result = motion_control_port_->StopAxis(axis);
            if (auto now = std::chrono::steady_clock::now();
                now - start > std::chrono::milliseconds(100)) {
                break;  // 超时保护
            }
        }
    }
}
